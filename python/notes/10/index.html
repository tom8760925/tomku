線性模型(Linear Models)

計算y = ax + b，計算出點的中心線

import matplotlib.pyplot as plt 
import numpy as np              
from sklearn.datasets import make_regression 
from sklearn.linear_model import LinearRegression 
from sklearn.model_selection import train_test_split 

# 資料集
X, y = make_regression(
    n_samples=100, # 生成100的點
    n_features=1, # 1個特徵
    n_informative=1, # 使用1個特徵
    noise=20, # 雜訊值
    random_state=1 # 隨機種子
)

# 將資料集分為訓練集和驗證集以7：3分
X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.3, random_state=1)

model = LinearRegression() # 訓練y = ax + b
model.fit(X_train, y_train) # 訓練

# 顯示訓練集的點
plt.scatter(X_train, y_train, s=60, color='blue')
# 顯示驗證集的點
plt.scatter(X_test, y_test, s=60,  color='green')

# 取得最小值到最大值的500的點一維之後，變成一個點一維的二維
X_plot = np.linspace(X.min(), X.max(), 500).reshape(-1, 1)
# 根據訓練的取得y
y_plot_pred = model.predict(X_plot)

# 畫線
plt.plot(X_plot, y_plot_pred, color='red')
# 顯示
plt.show()

線性和二次判別分析(Linear and Quadratic Discriminant Analysis)

用直線和曲線畫分類

import matplotlib.pyplot as plt
import numpy as np
from sklearn.datasets import make_classification
from sklearn.discriminant_analysis import LinearDiscriminantAnalysis, QuadraticDiscriminantAnalysis
from sklearn.model_selection import train_test_split
from sklearn.metrics import accuracy_score
from matplotlib.colors import ListedColormap

# 資料集
X, y = make_classification(
    n_samples=200,      # 200 個樣本
    n_features=2,       # 2 個特徵
    n_informative=2,    # 使用2個特徵
    n_redundant=0,      # 沒有冗餘特徵
    n_clusters_per_class=1, # 每個類別只有一個簇
    random_state=1     # 固定隨機種子
)

# 將資料集分為訓練集和驗證集以7：3分
X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.3, random_state=1)

# 設定顏色
cmap_bold = ListedColormap(['#FF0000', '#0000FF'])
cmap_light = ListedColormap(['#FFAAAA', '#AAAAFF']) 

# 訓練集1和2
# 訓練集最小值到最大值之間以0.02區間生成點
X_a = np.arange(X_train[:, 0].min() - 1, X_train[:, 0].max() + 1, 0.02)
Y_a = np.arange(X_train[:, 1].min() - 1, X_train[:, 1].max() + 1, 0.02)
# 轉換
# 將兩個1維轉換2維
# x[1,2],y[3,4] -> xx[[1,2],[1,2]],yy[[3,3],[4,4]]
xx, yy = np.meshgrid(X_a,Y_a)

# 訓練模型
lda_model = LinearDiscriminantAnalysis()
lda_model.fit(X_train, y_train)

# 根據訓練的取得z
# 先將2維轉1維，再將兩個1維合併2維
# xx[[1,2],[1,2]],yy[[3,3],[4,4]] -> [1,3],[1,4],[2,3],[2,4]
Z = lda_model.predict(np.c_[xx.ravel(), yy.ravel()])
# 轉換
# 根據範圍將1維轉換2維
Z = Z.reshape(xx.shape)

# 畫範圍
plt.pcolormesh(xx, yy, Z, cmap=cmap_light, shading='auto')

# 畫點
plt.scatter(X_train[:, 0], X_train[:, 1], c=y_train, cmap=cmap_bold,s=60, alpha=0.8, edgecolors='k')
plt.scatter(X_test[:, 0], X_test[:, 1], c=y_test, cmap=cmap_bold,s=60, alpha=0.8, edgecolors='w', linewidth=1.5)

plt.show()

X_plot = np.linspace(X.min(), X.max(), 500).reshape(-1, 1)
# 根據訓練的取得y
y_plot_pred = model.predict(X_plot)

# 畫線
plt.plot(X_plot, y_plot_pred, color='red')
# 顯示
plt.show()
