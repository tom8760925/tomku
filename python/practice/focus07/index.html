2025-08-29

重點07-set

這篇是python的練習實作第focus07篇

python的set有不重複和不能用切片

# set
print("\n======set宣告======")
print("直接宣告",{1,2,3,4})
seta = {1,2,3,4}
print("變數宣告",set )
setb = seta
print("b指定a",setb)
setc = seta.copy()
print("c複製a",setc)
print("\n======set使用======")
setaa = {3,4,5,6}
print("set聯集",seta|setaa)
print("set交集",seta&setaa)
print("set差集",seta-setaa)
print("set對稱差集",seta^setaa)

輸出

======set宣告======
直接宣告 {1, 2, 3, 4}
變數宣告 <class 'set'>
b指定a {1, 2, 3, 4}
c複製a {1, 2, 3, 4}

======set使用======
set聯集 {1, 2, 3, 4, 5, 6}
set交集 {3, 4}
set差集 {1, 2}
set對稱差集 {1, 2, 5, 6}
