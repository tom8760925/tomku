C語言的練習-if

2024-03-03

if

這篇是C的練習筆記第05篇
C的if寫法是

#include <stdio.h>
#include <stdbool.h>
int main(){
    bool p = true;
    if(p){
        printf("p:%s\n",p?"true":"fales");
    }
    else{
        printf("p:%s\n",p?"true":"fales");
    }
    return 0;
}

結果

p:true

if也可以寫成這樣

#include <stdio.h>
#include <stdbool.h>
int main(){
    bool p = false;
    p = (2>1)?true:false;
    printf("p:%s\n",p?"true":"fales");
    return 0;
}

結果

p:true

多條件if

#include <stdio.h>
int main(){
    int a=1,b=2;
    if(a<b){
        printf("p:%d<%d\n",a,b);
    }
    else if(a>b)
    {
        printf("p:%d>%d\n",a,b);
    }
    else{
        printf("p:%d=%d\n",a,b);
    }
    return 0;
}

結果

p:1<2

多條件還有

#include <stdio.h>
int main(){
    char n='a';
    switch (n)
    {
    case 'a':
        printf("p:apple\n");
        break;
    case 'b':
        printf("p:banana\n");
        break;
    default:
        printf("p:error\n");
        break;
    }
    return 0;
}

結果

p:apple
