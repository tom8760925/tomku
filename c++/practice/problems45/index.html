重點

表格要和在一起看，例如第一行

[0][0] 會等於0是因為相等所以不用動
[0][1] 會等於1是因為""要變成a，所以新增a
[0][1] 會等於2是因為""要變成ab，所以新增a又新增b
[1][1] 會等於1是因為c要變成a，所以c變更a
[2][2] 會等於2是因為cb要變成ab，所以c變成a和d變成b

所以std::min({dp[i][j - 1], dp[i - 1][j], dp[i - 1][j - 1]})代表新增、刪除、修改

dp[i][j - 1]代表新增，dp[2][2]舉例，dp[2][1]那格已經是cd變成a，那到這格就是再新增b後ab，所以步數變成3
dp[i - 1][j]代表刪除，dp[2][2]舉例，dp[1][2]那格已經是c變成ab，那到這格就是多了d變成abd刪掉d後ab，所以步數變成3
dp[i - 1][j - 1]代表修改，dp[2][2]舉例，dp[1][1]那格已經是c變成a，那到這格就是多了d變成ad修改d後ab，所以步數變成2

結果

#include <iostream>
#include <vector>
#include <string>

class Solution
{
public:
    int minDistance(std::string word1, std::string word2)
    {
        int m = word1.length();
        int n = word2.length();
        std::vector<std::vector<int>> dp(m + 1, std::vector<int>(n + 1));

        for (int i = 0; i <= m; i++)
        {
            dp[i][0] = i;
        }
        for (int i = 0; i <= n; i++)
        {
            dp[0][i] = i;
        }
        for (int i = 1; i <= m; i++)
        {
            for (int j = 1; j <= n; j++)
            {
                if (word1[i - 1] == word2[j - 1])
                {
                    dp[i][j] = dp[i - 1][j - 1];
                }
                else
                {
                    dp[i][j] = 1 + std::min({dp[i][j - 1], dp[i - 1][j], dp[i - 1][j - 1]});
                }
            }
        }
        return dp[m][n];
    }
};

int main()
{
    Solution s;
    std::cout << s.minDistance("horse", "ros") << std::endl;
    return 0;
}
