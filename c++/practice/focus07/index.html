C++語言的練習-重點練習07

這篇是C++練習-重點練習07篇

++的重點

#include <iostream>

int main()
{
    int i = 1;

    // 後做
    std::cout << "後做\n";
    // 會先執行cout<<i，後在執行i+1
    std::cout << "i:" << i++ << '\n';
    std::cout << "i:" << i << '\n';

    std::cout << "-------\n";

    // 前做
    std::cout << "前做\n";
    // 會先執行i+1，後執行cout<<i
    std::cout << "i:" << ++i << '\n';
    std::cout << "i:" << i << '\n';

    std::cout << std::endl;

    return 0;
}

結果

後做
i:1
i:2
-------
前做
i:3
i:3
