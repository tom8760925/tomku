Find Minimum in Rotated Sorted Array

這篇是C++的練習實作第problems36篇

練習leetcode的Find Minimum in Rotated Sorted Array

解法

排列

排列後就能找到最小

class Solution
{
public:
    int findMin(std::vector<int> &nums)
    {
        int r = 0;
        if (nums.size() == 0)
        {
            return 0;
        }
        for (int n = 0; n < int(nums.size()) - 1; n++)
        {
            for (int nn = n + 1; nn < int(nums.size()); nn++)
            {
                if (nums[n] > nums[nn])
                {
                    r = nums[n];
                    nums[n] = nums[nn];
                    nums[nn] = r;
                }
            }
        }
        return nums[0];
    }
};

二元收尋

二元收尋找到最小值
先比較中間值和最右側大小，如果中間值比較大，就代表最小值在右側，接著將l搬到中間值右側，接著重複直到找到值為止。

class Solution2
{
public:
    int findMin(std::vector<int> &nums)
    {
        int l = 0;
        int r = nums.size() - 1;

        while (l < r)
        {
            int m = l + (r - l) / 2;
            if (nums[m] <= nums[r])
            {
                r = m;
            }
            else
            {
                l = m + 1;
            }
        }

        return nums[l];
    }
};
