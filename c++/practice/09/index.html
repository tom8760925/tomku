C++語言的練習-練習實作例外

2024-12-20

練習實作例外

這篇是C++的練習實作第09篇

try

例外使用上有幾個重點：

  1. 不能亂用，有需要再使用，而且不是作為判斷式使用。
  2. 要使用就不能只是顯示錯誤訊息，要做些什麼，例如exit。
  3. 錯誤訊息要可以讓人知道，是哪裡發生錯誤，發生了什麼錯誤。
  4. 不能只使用std::exception，因為這樣會缺乏錯誤的資訊和不太方便後續的處理，但可以當作錯誤偵測，之後發現有新的錯誤時，再新增例外處理。

1.cpp

#include "2.hpp"

int main()
{
    try
    {
        int n = 0;
        if (n == 0)
        {
            throw outerr::NotExitException(1, "n=0", "main");
        }
    }
    catch (outerr::NotExitException &e)
    {
        // 顯示錯誤訊息
        std::cerr << e.what() << std::endl;
        //後續的程式
    }
    catch (const std::exception &e)
    {
        std::cerr << e.what() << std::endl;
        exit(1);
    }

    try
    {
        int n = 0;
        if (n == 0)
        {
            throw outerr::ErrException(1, "n=0", "main");
        }
    }
    catch (outerr::ErrException &e)
    {
        // 顯示錯誤訊息
        std::cerr << e.what() << std::endl;
        exit(1);
    }
    catch (const std::exception &e)
    {
        std::cerr << e.what() << std::endl;
        exit(1);
    }

    return 0;
}

2.hpp

#ifndef try1_hpp
#define try1_hpp

#include <iostream>
#include <stdio.h>

namespace outerr
{
    class ErrException : public std::exception
    {
    public:
        ErrException(int ID, const char *err, const char *func);
        const char *what() const noexcept override;

    private:
        int ID;
        const char *err;
        const char *func;
    };
    class NotExitException : public std::exception
    {
    public:
        NotExitException(int ID, const char *err, const char *func);
        const char *what() const noexcept override;

    private:
        int ID;
        const char *err;
        const char *func;
    };
}

#endif

2.cpp

#include "2.hpp"

outerr::ErrException::ErrException(int ID, const char *err, const char *func)
{
    this->ID = ID;
    this->err = err;
    this->func = func;
}

const char *outerr::ErrException::what() const noexcept
{
    std::string str = "";
    str.append("ID: [");
    str.append(std::to_string(this->ID));
    str.append("] ErrException: [");
    str.append(err);
    str.append("] Function: [");
    str.append(this->func);
    str.append("]");
    char *r = new char[strlen(str.c_str()) + 1];
    strlcpy(r, str.c_str(), strlen(str.c_str()) + 1);
    return r;
}

outerr::NotExitException::NotExitException(int ID, const char *err, const char *func)
{
    this->ID = ID;
    this->err = err;
    this->func = func;
}

const char *outerr::NotExitException::what() const noexcept
{
    std::string str = "";
    str.append("ID: [");
    str.append(std::to_string(this->ID));
    str.append("] NotExitException: [");
    str.append(err);
    str.append("] Function: [");
    str.append(this->func);
    str.append("]");
    char *r = new char[strlen(str.c_str()) + 1];
    strlcpy(r, str.c_str(), strlen(str.c_str()) + 1);
    return r;
}
