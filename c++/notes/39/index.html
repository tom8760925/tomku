C++語言的筆記-operator

2025-01-07

operator

這篇是C++的筆記第39篇

operator 是運算重載，是可以更改運算。

加法

#include <iostream>

class a
{
public:
    int n1 = 0, n2 = 0;
    a();
    a(int, int);

    a operator+(a &_a)
    {
        this->n1 += _a.n1;
        this->n2 += _a.n2;
        a ra(this->n1, this->n2);
        return ra;
    }
    int getsun();
};
a::a()
{
}
a::a(int n1, int n2)
{
    this->n1 = n1;
    this->n2 = n2;
}
int a::getsun()
{
    return this->n1 + this->n2;
}
int main()
{
    a a1(2, 3), a2(4, 5), a3;
    a3 = a1 + a2;
    std::cout << a3.getsun() << std::endl;
    return 0;
}

#include <iostream>

class a
{
public:
    a(int n);
    int operator+();

private:
    int n = 0;
};
a::a(int n)
{
    this->n = n;
}
int a::operator+()
{
    return this->n;
}

int main()
{
    a _a1(3);

    int n = 1;
    n = n + _a1.operator+();
    std::cout << n << std::endl;

    return 0;
}

bool

#include <iostream>

class a
{
public:
    a(int n);
    operator bool()
    {
        if (n == 0)
        {
            return true;
        }
        else
        {
            return false;
        }
    }

private:
    int n = 0;
};
a::a(int n)
{
    this->n = n;
}

int main()
{
    a _a1(0), _a2(1);

    std::cout << _a1 << std::endl;
    std::cout << _a2 << std::endl;
    return 0;
}

轉型態或輸出職

#include <iostream>

class a
{
public:
    a(int n);
    operator int();
    int getn();

private:
    int n = 0;
};
a::a(int n)
{
    this->n = n;
}
a::operator int()
{
    return this->n;
}
int a::getn()
{
    return this->n;
}

int main()
{
    a _a1(3);
    int n = 1;
    n += _a1;
    std::cout << n << std::endl;

    return 0;
}
