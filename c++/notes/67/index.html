Type Erasure#

將不同物件組合並可擴展，也減少使用多型，並也減少相同程式，也將指數轉換成值的使用。

---
title: Observer
config:
  theme: dark
  class:
    hideEmptyMembersBox: true
---
classDiagram
class Sports{
    -int Speed;
    +Sports(int Speed)
    +void GetSpeed()
}
class Bus{
    -int Speed;
    +Bus(int Speed)
    +void GetSpeed()
}
class DetailCarConcept["Detail::CarConcept"]{
    <<interface>>
    +virtual void GetSpeed() const = 0;
    +virtual std::unique_ptr~CarConcept~ Close() const = 0;
}
class DetailCarModel["Detail::CarModel"]{
    -typename CarT _CarT;
    -typename SpeedStrategyT _SpeedStrategyT;
    +CarModel(typename CarT _CarT, typename SpeedStrategyT _SpeedStrategyT)
    +void GetSpeed() const override
    +std::unique_ptr~CarConcept~ Close() const override
}
DetailCarConcept<|..DetailCarModel
class Car{
    -friend void GetSpeed(Car &_Car)
    -std::unique_ptr~Detail::CarConcept~ pimpl
    +Car(typename CarT _CarT, typename SpeedStrategyT _SpeedStrategyT)
    +Car(Car &_Car) : pimpl(_Car.pimpl->Close())
    +Car &operator=(Car &_Car)
}
Car-->DetailCarModel
Car-->Sports
Car-->Bus
