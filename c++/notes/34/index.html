C++語言的筆記-互相include

2024-11-03

互相include

這篇是C++的筆記第34篇

當兩個標頭檔需要互相使用時
使用下面方法執行

1.hpp

#ifndef a_hpp
#define a_hpp

#include "2.hpp"

#include <iostream>

class a
{
public:
    void aa();
    void ab();

private:
    b *_b;
};

#endif

1.cpp

#include "1.hpp"

void a::aa()
{
    std::cout << "aa" << std::endl;
}

void a::ab()
{
    _b = new b();
    _b->bb();
}

2.hpp

#ifndef b_hpp
#define b_hpp

#include <iostream>

class a;

class b
{
public:
    void bb();
    void ba();

private:
    a *_a;
};

#endif

2.cpp

#include "2.hpp"
#include "1.hpp"

void b::bb()
{
    std::cout << "bb" << std::endl;
}

void b::ba()
{
    _a = new a();
    _a->aa();
}

3.cpp

#include <iostream>

#include "1.hpp"

int main(int argc, const char *argv[])
{

    a _a;

    _a.aa();
    _a.ab();

    std::cout << "-------\n";
    b _b;

    _b.bb();
    _b.ba();

    return 0;
}
