mysql的筆記-指令

2024-04-21

指令

這篇是mysql的筆記第02篇
記錄mysql指令

資料庫

CREATE DATABASE `test` ;//新增資料庫
DROP DATABASE `test`; //刪除資料庫
SHOW DATABASES;//顯示資料庫
use test;//指定使用的資料庫

資料表

CREATE TABLE tests (新增資料表
    id INT PRIMARY KEY NOT NULL UNIQUE AUTO_INCREMENT, //新增主鍵、自動填值、不能重複、不能為null
    v VARCHAR(50) NOT NULL,//不能為null
    i int NOT NULL CHECK (i>0)//不能為null、值要大於0
);
ALTER TABLE tests ADD ii int not null;//修改新增欄位
ALTER TABLE tests ADD FOREIGN KEY (test_id) REFERENCES test(id);//建立外鍵，這一行只能新增關聯表格的那一行
ALTER TABLE test DROP FOREIGN KEY ibfk;//移除外鍵，要使用外鍵的名稱
ALTER TABLE test AUTO_INCREMENT=1 //重置AUTO_INCREMENT
INSERT INTO tests VALUES (1,"A",1,1);//新增資料
INSERT INTO tests (v,i,ii) VALUES ("A",1,1);//新增資料夾指定欄位，而且id欄位有加AUTO_INCREMENT，所以可以省略
Insert into tests(i,v) VALUES(1,(select id from tests2 where id = 1)) ;//可以依照其他表格的資料，新增表格資料
UPDATE tests SET ii=3 WHERE id=4;//更新資料
DELETE FROM tests WHERE ID=2;//刪除資料
DROP TABLE `test`.`tests`;//刪除資料表
ALTER TABLE tests DRop vv;//刪除欄位
SELECT * FROM tests;顯示資料表資料
SELECT t.Id,tt.Name , t.Data FROM test as t INNER JOIN test2 AS tt ON t.test2_Id = tt.Id;//關聯查詢
SELECT t.Id,tt.Name , t.Data FROM test as t INNER JOIN (SELECT Id, Name FROM test2) AS tt ON t.test2_Id = tt.Id;//關聯查詢

函數

GROUP_CONCAT(t.data)//根據某一行將資料組合，需要使用 GROUP BY 指定參考行
GROUP BY t.Id //群組
CONCAT('Id:',t.Id)//指定輸出內容
if(t.Id = 0,'Yes','No')//判斷
t.data like ('%a%') //收尋資料字串是否有a

其他

mysql有個安全機制，如果沒有使使用where的情況使用UPDATE或DELETE FROM，會阻擋使用，可以設定關閉

SET SQL_SAFE_UPDATES=0
