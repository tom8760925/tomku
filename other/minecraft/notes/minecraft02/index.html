minecraft mod 設定#

這篇是minecraft mod 設定的筆記

mods.toml 模組註冊表單

modLoader = "javafml" #登入器
loaderVersion = "${loader_version_range}" # forge版本
license = "${mod_license}" # 許可證
[[mods]]
modId = "${mod_id}" # id
version = "${mod_version}" # mc版本
displayName = "${mod_name}" # 名子
authors = "${mod_authors}" # 作者
description = '''${mod_description}''' # 簡介

[[dependencies.${mod_id}]] #需求依賴
    modId="forge" 
    mandatory=true 
    versionRange="${forge_version_range}"
    ordering="NONE"
    side="BOTH"
[[dependencies.${mod_id}]] #需求依賴
    modId="minecraft"
    mandatory=true
    versionRange="${minecraft_version_range}"
    ordering="NONE"
    side="BOTH"

build.gradle 環境設定檔

plugins { //插件
    id 'eclipse'
    id 'maven-publish'
    id 'net.minecraftforge.gradle' version '6.+'
}
version = mod_version // mc版本
group = mod_group_id 
base {
    archivesName = mod_id // id
}
// java版本
java.toolchain.languageVersion = JavaLanguageVersion.of(17) 
//測試平台
minecraft { 
    mappings channel: mapping_channel, version: mapping_version
    runs {
        configureEach {
            workingDirectory project.file('run')
            property 'forge.logging.markers', 'REGISTRIES'
            property 'forge.logging.console.level', 'debug'

            mods {
                "${mod_id}" {
                    source sourceSets.main
                }
            }
        }
        client {
            property 'forge.enabledGameTestNamespaces', mod_id
        }

        server {
            property 'forge.enabledGameTestNamespaces', mod_id
            args '--nogui'
        }
        gameTestServer {
            property 'forge.enabledGameTestNamespaces', mod_id
        }

        data {
            workingDirectory project.file('run-data')
            args '--mod', mod_id, '--all', '--output', file('src/generated/resources/'), '--existing', file('src/main/resources/')
        }
    }
}
sourceSets.main.resources { srcDir 'src/generated/resources' } // 載入的資源
// 依賴庫
repositories { 

}
//依賴的模組
dependencies { 
    minecraft "net.minecraftforge:forge:${minecraft_version}-${forge_version}"
}
tasks.named('processResources', ProcessResources).configure { //轉換字串
    var replaceProperties = [
            minecraft_version: minecraft_version, minecraft_version_range: minecraft_version_range,
            forge_version: forge_version, forge_version_range: forge_version_range,
            loader_version_range: loader_version_range,
            mod_id: mod_id, mod_name: mod_name, mod_license: mod_license, mod_version: mod_version,
            mod_authors: mod_authors, mod_description: mod_description,
    ]
    inputs.properties replaceProperties

    filesMatching(['META-INF/mods.toml', 'pack.mcmeta']) {
        expand replaceProperties + [project: project]
    }
}
// 屬性
tasks.named('jar', Jar).configure {
    manifest {
        attributes([
                'Specification-Title'     : mod_id,
                'Specification-Vendor'    : mod_authors,
                'Specification-Version'   : '1',
                'Implementation-Title'    : project.name,
                'Implementation-Version'  : project.jar.archiveVersion,
                'Implementation-Vendor'   : mod_authors,
                'Implementation-Timestamp': new Date().format("yyyy-MM-dd'T'HH:mm:ssZ")
        ])
    }
    finalizedBy 'reobfJar'
}
// 使用maven-publish
publishing {
    publications {
        register('mavenJava', MavenPublication) {
            artifact jar
        }
    }
    repositories {
        maven {
            url "file://${project.projectDir}/mcmodsrepo"
        }
    }
}
// 使用utf-8進行編譯
tasks.withType(JavaCompile).configureEach {
    options.encoding = 'UTF-8' // Use the UTF-8 charset for Java compilation
}
