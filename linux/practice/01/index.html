linux的練習-練習實作-判斷是否為path底下

2025-04-23

判斷是否為path底下

這篇是linux的練習實作第01篇

#!/bin/sh

Path_s()
{
var=$1 r=0

# 將字串轉成陣列，依據：分開
IFS=':' read -a arr <<< "$PATH"

# 陣列迴圈
for n in "${arr[@]}"
do
    if [ -x "$n/$var" ]
    then
        r=1
    fi
done

# 回傳
return $r
}

Path_s $1

# 顯示回傳值
echo "$?"
