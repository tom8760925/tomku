Qt widget切換和傳值

這篇是Qt widget切換和傳值的筆記

mainwindow和widgetwindow的widget介面切換

mainwindow.h

#ifndef MAINWINDOW_H
#define MAINWINDOW_H

#include <QMainWindow>

QT_BEGIN_NAMESPACE
namespace Ui {
class MainWindow;
}
QT_END_NAMESPACE

class MainWindow : public QMainWindow
{
    Q_OBJECT

public:
    MainWindow(QWidget *parent = nullptr);
    ~MainWindow();

private slots:
    void on_widgetButton_clicked();

private:
    Ui::MainWindow *ui;
};
#endif // MAINWINDOW_H

mainwindow.cpp

#include "mainwindow.h"
#include "ui_mainwindow.h"
#include "widgetwindow.h"


MainWindow::MainWindow(QWidget *parent)
    : QMainWindow(parent)
    , ui(new Ui::MainWindow)
{
    ui->setupUi(this);
}

MainWindow::~MainWindow()
{
    delete ui;
}

void MainWindow::on_widgetButton_clicked()
{
    //宣告新建的widget
    widgetwindow *_widgetwindow = new widgetwindow(nullptr,ui,this);

    //隱藏現在的UI
    ui->centralwidget->setVisible(false);

    //清空CentralWidget
    this->takeCentralWidget();

    //設定CentralWidget
    this->setCentralWidget(_widgetwindow);

}

widgetwindow.h

#ifndef WIDGETWINDOW_H
#define WIDGETWINDOW_H

#include <QWidget>
#include "ui_mainwindow.h"
#include "mainwindow.h"

namespace Ui {
class widgetwindow;
}

class widgetwindow : public QWidget
{
    Q_OBJECT

public:
    explicit widgetwindow(QWidget *parent = nullptr, Ui::MainWindow *uimain = nullptr, MainWindow *mainw = nullptr);
    ~widgetwindow();

private slots:
    void on_mainButton_clicked();

    void on_windowButton_clicked();

private:
    Ui::widgetwindow *ui;
    Ui::MainWindow *uimain;
    MainWindow *mainw;
};

#endif // WIDGETWINDOW_H

widgetwindow.cpp

#include "widgetwindow.h"
#include "ui_widgetwindow.h"
#include "ui_mainwindow.h"


widgetwindow::widgetwindow(QWidget *parent, Ui::MainWindow *uimain, MainWindow *mainw)
    : QWidget(parent)
    , ui(new Ui::widgetwindow)
{
    ui->setupUi(this);
    this->uimain = uimain;
    this->mainw = mainw;
}

widgetwindow::~widgetwindow()
{
    delete ui;
}

void widgetwindow::on_mainButton_clicked()
{
    //取得MainWindow的textEdit，設定widgetwindow的label
    ui->labelmain->setText(uimain->textEditwidget->toPlainText());

}


void widgetwindow::on_windowButton_clicked()
{
    //取得widgetwindow的textEdit，設定MainWindow的label
    uimain->labelwidget->setText(ui->textEditwidget->toPlainText());

    //關閉widgetwindow
    this->close();

    //顯示MainWindow的UI
    uimain->centralwidget->setVisible(true);

    //清空CentralWidget
    mainw->takeCentralWidget();

    //設定CentralWidget
    mainw->setCentralWidget(uimain->centralwidget);


}
