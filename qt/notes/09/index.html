這篇是Qt QTableWidget的筆記

QTableWidget

使用方法。

#include <QTableWidget>
// 宣告
QTableWidget *_QTableWidget = new QTableWidget();
//設定啟用
_QTableWidget->setEnabled();
// 輸出文字
_QTableWidget->Item();
// 新增文字
_QTableWidget->setItem();
// 清除指定的資料
_QTableWidget->takeItem();
// 輸出widget
_QTableWidget->CellWidget();
// 新增widget
_QTableWidget->setCellWidget();
// 清除指定的widget
_QTableWidget->removeCellWidget();
// 清除指定的行
_QTableWidget->removeRow();
// 設定最小高度
_QTableWidget->setMinimumHeight();
// 設定最小寬度
_QTableWidget->setMinimumWidth();
// 設定最大寬度
_QTableWidget->setMaximumWidth();
// 設定樣板
_QTableWidget->setStyleSheet();
// 新增事件
_QTableWidget->connect():
// 數量
_QTableWidget->count();
// 行數量
_QTableWidget->rowCount();
// 設定表格行欄位數
_QTableWidget->setrowCount();
// 列數量
_QTableWidget->ColumnCount();
// 設定表格列欄位數
_QTableWidget->setColumnCount();
// 清掉整個
_QTableWidget->clear();
// 清掉內容
_QTableWidget->clearContents();
// 清除
_QTableWidget->close();
// 輸出選取的行
_QTableWidget->currentRow();
// 輸出選取的列
_QTableWidget->currentColumn();
// 設定選取的格
_QTableWidget->setCurrentCell();
// 設定關閉行點選
_QTableWidget->verticalHeader()->setSectionsClickable(false);
// 設定行欄位寬度
_QTableWidget->verticalHeader()->setDefaultSectionSize();
// 設定關閉列點選
_QTableWidget->horizontalHeader()->setSectionsClickable(false);
// 設定列欄位寬度
_QTableWidget->horizontalHeader()->setDefaultSectionSize();
// 設定關閉左上角按鈕啟用
_QTableWidget->setCornerButtonEnabled(false);
// 設定滾輪顯示
_QTableWidget->setVerticalScrollBarPolicy(Qt::ScrollBarAlwaysOn);
// 設定行名稱
_QTableWidget->setHorizontalHeaderLabels();
// 設定點選一行
_QTableWidget->setSelectionBehavior(QTableWidget::SelectRows);
